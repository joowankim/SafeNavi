217  private final BroadcastReceiver mReceiver = new BroadcastReceiver() {
218  @Override
219  public void onReceive(Context context, Intent intent) {
220  String action = intent.getAction();
221 
222  // When discovery finds a device
223  if (BluetoothDevice.ACTION_FOUND.equals(action)) {
224  // Get the BluetoothDevice object from the Intent
225  BluetoothDevice device = intent.getParcelableExtra(BluetoothDevice.EXTRA_DEVICE);
226  // If it's already paired, skip it, because it's been listed already
227  if (device.getBondState() != BluetoothDevice.BOND_BONDED) {
228  mNewDevicesArrayAdapter.add(device.getName() + "\n" + device.getAddress());
229  }
230  // When discovery is finished, change the Activity title
231  } else if (BluetoothAdapter.ACTION_DISCOVERY_FINISHED.equals(action)) {
232  setProgressBarIndeterminateVisibility(false);
233  setTitle(R.string.select_device);
234  if (mNewDevicesArrayAdapter.getCount() == 0) {
235  String noDevices = getResources().getText(R.string.none_found).toString();
236  mNewDevicesArrayAdapter.add(noDevices);
237  }
